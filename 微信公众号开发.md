              data = web.input()
              if len(data) == 0:

              signature = data.signature
              timestamp = data.timestamp
              nonce = data.nonce

              list.sort()
              tmp_str = ''.join(list)
              hashcode = hashlib.sha1(tmp_str.encode('utf-8')).hexdigest()
